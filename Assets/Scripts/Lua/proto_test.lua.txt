local pb = require 'pb'
local protoc = require 'protoc'

file = io.open("Assets/Scripts/Lua/person_out.proto");

message = '';

line = file:read();

while(line ~= nil) do
    message = string.format("%s %s",message,line);
    line = file:read();
end

assert(protoc:load(message));
