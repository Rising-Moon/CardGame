local object = class


function object


function object:new(obj)
    obj=obj or {}
    setmetatable(obj,self)
    self.__index=self
    --时间取反再取要7位数值，设为随机数因子
    math.randomseed(tostring(os.time()):reverse():sub(1, 7))
    self.objId=math.random(1,100) or {} --产生1到100之间的随机数
    return obj

end

return object
