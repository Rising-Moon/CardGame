--创建一个table
local  Object = {objname =nil };



-- 构造对象的方法
function Object:new(objname)
    --创建一个对象table
    local instance = {};
    instance.objname=objname;
    --设置元表                                             
    setmetatable(instance,{__index = self});
    return instance
end

--添加一个方法
function Object:change(objname)

    print(self.objname);
    print("tree:grow");
    self.objname=objname;
    print(self.objname)
end

return Object
