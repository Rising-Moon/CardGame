local resourcesManager ={}

function resourcesManager:init()
    print("this is init");
end

function resourcesManager:loadAllTypeResources(path,resType,seltype,position,abName)
    local go = {};
    --1 resources
    if resType ==1 then
        go =CS.UnityEngine.Object.Instantiate(path,position)
    elseif resType==2 then
        print()
    else
        print('other type need to be');
    end
    return go
end

function resourcesManager:resourcesLoad(path,seltype)
    local res ={};

    res = CS.UnityEngine.Object.Resources.Load<seltype>("path");
    return res

end
--异步加载AB包
function resourcesManager:assetBundle(path,seltype,abName)
    local res ={};
    res =  CS.UnityEngine.Object.AssetBundle.LoadFrom(path);

end
